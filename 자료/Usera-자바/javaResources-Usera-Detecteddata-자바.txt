public class Detecteddata {
	private String num;
	private String gender;
	private String age;
	public String getNum() {
		return num;
	}
	public void setNum(String num) {
		this.num = num;
	}

		this.gender = gender;
	}
	public String getAge() {
		return age;
	}
	public void setAge(String age) {
		this.age = age;
	}
	
}
